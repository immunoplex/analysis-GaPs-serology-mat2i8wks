VISIT_PRETTY <- c(PregEarly = "early", MatBirth = "birth")
pretty_feature <- function(f) {
  base <- sub("@.*$", "", f); visit <- sub("^.*@", "", f)
  vp <- ifelse(visit %in% names(VISIT_PRETTY), VISIT_PRETTY[visit], visit)
  paste0(gsub("_", " ", base), " @ ", vp)
}
prep_features <- function(su, keep = NULL, top_n = NULL, baseline_visit = params$baseline_visit) {
  su$base_feature <- sub("@.*$", "", su$feature); su$visit <- sub("^.*@", "", su$feature)
  cls <- classify_feature(unique(su$base_feature))
  su <- dplyr::left_join(su, cls, by = c("base_feature" = "feature"))
  if (!is.null(keep)) { su <- su[match(keep, su$feature), , drop = FALSE]
    su <- su[!is.na(su$feature), , drop = FALSE]
  } else { su <- su[order(-su$auc), , drop = FALSE]; if (!is.null(top_n)) su <- utils::head(su, top_n) }
  su$antigen <- sub("^IPV[0-9]+$", "IPV", su$antigen)
  su$antigen <- ifelse(su$antigen %in% names(ANTIGEN_COL), su$antigen, "DEFAULT")
  su$prevax <- su$visit == baseline_visit; su$plabel <- pretty_feature(su$feature)
  su$plabel <- factor(su$plabel, levels = rev(su$plabel)); su
}
use_pattern <- requireNamespace("ggpattern", quietly = TRUE)
signature_panel <- function(df, title, show_legend = FALSE) {
  ANT_BREAKS <- c("PRN","PT","FHA","FIM","IPV","DT","TT")
  fill_guide <- if (show_legend) guide_legend(order = 1, nrow = 1) else "none"
  mark_guide <- if (show_legend) guide_legend(order = 2, nrow = 1) else "none"
  p <- ggplot(df, aes(auc, plabel, fill = antigen))
  if (use_pattern) {
    p <- p + ggpattern::geom_col_pattern(aes(pattern = prevax), width = 0.72, colour = NA,
      pattern_fill = "white", pattern_angle = 45, pattern_density = 0.12,
      pattern_spacing = 0.03, pattern_size = 0.05) +
      ggpattern::scale_pattern_manual(values = c(`TRUE` = "stripe", `FALSE` = "none"),
        breaks = TRUE, labels = c(`TRUE` = "pre-vaccination"), name = NULL, guide = mark_guide)
  } else {
    p <- p + geom_col(aes(colour = prevax), width = 0.72, linewidth = 0.6) +
      scale_colour_manual(values = c(`TRUE` = "grey15", `FALSE` = NA),
        breaks = TRUE, labels = c(`TRUE` = "pre-vaccination"), name = NULL, guide = mark_guide)
  }
  df$inside <- df$auc >= 0.72
  p + geom_text(data = df[df$inside, , drop = FALSE], aes(label = sprintf("%.2f", auc)),
        hjust = 1.15, colour = "white", fontface = "bold", size = 3.1) +
    geom_text(data = df[!df$inside, , drop = FALSE], aes(label = sprintf("%.2f", auc)),
        hjust = -0.20, colour = "grey20", fontface = "bold", size = 3.1) +
    scale_fill_manual(values = ANTIGEN_COL, limits = ANT_BREAKS, breaks = ANT_BREAKS,
        na.value = "#9a9a9a", name = NULL, guide = fill_guide) +
    scale_x_continuous(breaks = seq(0.5, 1.0, 0.1), expand = c(0, 0)) +
    coord_cartesian(xlim = c(0.5, 1.0)) +
    labs(x = "Univariate AUC (Low vs High responder)", y = NULL, title = title) +
    theme_minimal(base_size = 11) +
    theme(panel.grid.major.y = element_blank(), panel.grid.minor = element_blank(),
          plot.title = element_text(face = "bold", size = 12), axis.ticks = element_blank(),
          legend.position = if (show_legend) "bottom" else "none", legend.box = "horizontal",
          legend.key.size = grid::unit(12, "pt"), legend.text = element_text(size = 9))
}

NEW — Supplemental companion: interval robustness

This companion figure is the visual answer to the reviewer/coauthor concern that a mother’s delivery titre (half of the responder definition) is set by the vaccination-to-delivery interval. For each of the three score-defining antigens we plot delivery IgG against the interval (weeks), by arm, with the natural-spline fit used in Phase 10 (baseline-adjusted). Flat curves = the interval is not moving the titres that build the score.

arm_tbl$arm_name <- factor(as.character(arm_tbl$arm_name), levels = c("TT", "TdaP"))
interval_tbl <- make_interval_tbl(arm_tbl, clin, maternal_arms = c("TT", "TdaP"))
it_avail <- interval_tbl[is.finite(interval_tbl$w_weeks), , drop = FALSE]

paired <- paired_change(data_raw, params$from_visit, params$to_visit)
paired <- paired[paired$arm_name %in% c("TT", "TdaP"), , drop = FALSE]
paired$arm_name <- factor(as.character(paired$arm_name), levels = c("TT", "TdaP"))

DEF <- params$responder_features
ARM_COLS <- c(TdaP = "#8A1538", TT = "#2C7A7B")

# quantitative anchor: within-arm partial R^2 of the interval spline per antigen
s1 <- boost_spline_table(paired, DEF, spline_df = params$spline_df)

# combined curve + point frames across the three defining antigens
.cur <- list(); .pts <- list()
for (ft in DEF) {
  fit <- fit_boost_spline(paired[paired$feature == ft, , drop = FALSE], spline_df = params$spline_df)
  if (!isTRUE(fit$ok)) next
  cd <- spline_curve_df(fit); if (!is.null(cd)) { cd$feature <- ft; .cur[[ft]] <- cd }
  pt <- spline_points_df(fit); if (!is.null(pt)) { pt$feature <- ft; .pts[[ft]] <- pt }
}
curve_df <- do.call(rbind, .cur); point_df <- do.call(rbind, .pts)
lab_df <- data.frame(
  feature = s1$feature,
  lab = sprintf("partial R\u00b2 (w): TdaP %.2f | TT %.2f", s1$pR2_w_TdaP, s1$pR2_w_TT),
  stringsAsFactors = FALSE)
ord <- intersect(DEF, unique(curve_df$feature))
curve_df$feature <- factor(curve_df$feature, levels = ord)
point_df$feature <- factor(point_df$feature, levels = ord)
lab_df$feature   <- factor(lab_df$feature,   levels = ord)
p_iv <- ggplot() +
  geom_point(data = point_df, aes(w_weeks, adj, colour = arm_name), alpha = 0.30, size = 1.0) +

              alpha = 0.15, colour = NA) +
  geom_line(data = curve_df, aes(w_weeks, fit, colour = arm_name), linewidth = 0.9) +
  geom_text(data = lab_df, aes(x = -Inf, y = Inf, label = lab), hjust = -0.05, vjust = 1.4,
            size = 3.0, colour = "grey25") +

  scale_colour_manual(values = ARM_COLS, name = "Arm") +
  scale_fill_manual(values = ARM_COLS, guide = "none") +
  labs(x = "weeks: vaccination \u2192 delivery",

       title = "Interval robustness of the score-defining delivery titres") +

Figure S (interval robustness). Delivery (MatBirth) IgG for the three score-defining antigens against the vaccination-to-delivery interval (weeks), by maternal arm. Line = natural-spline fit at the median pre-vaccination baseline; band = 95% CI; points = per-mother values adjusted to that baseline (partial residuals). Within-arm partial R-squared of the interval spline is annotated per panel. The near-flat fits show the interval explains little of the delivery-titre variation that builds the responder score.
