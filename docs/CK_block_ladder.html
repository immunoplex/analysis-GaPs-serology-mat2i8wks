Data assembly

The modelling frame is identical to Results C–F, except the design/clinical base is now assembled from three named variable groups so the ladder can attribute predictability to design (vaccination), pre-vaccination clinical, and birth clinical separately. Antibody blocks (maternal production / cord transfer / concurrent) are unchanged.

ANTI_IGG <- params$anti_igg

pull_wide <- function(d, visit, feats, suffix) {
  d %>%
    dplyr::filter(visit_name == visit, feature %in% feats) %>%
    dplyr::distinct(subject_accession, feature, .keep_all = TRUE) %>%
    dplyr::select(subject_accession, feature, log_assay_value) %>%
    tidyr::pivot_wider(names_from = feature, values_from = log_assay_value) %>%
    dplyr::rename_with(~ paste0(.x, suffix), -subject_accession)
}

# base-R fit utilities (identical contract to Results C–F)
r2 <- function(form, data) tryCatch({
  cc <- data[stats::complete.cases(data[, all.vars(form), drop = FALSE]), , drop = FALSE]
  if (nrow(cc) < 5) return(NA_real_)
  summary(stats::lm(form, cc))$r.squared
}, error = function(e) NA_real_)

cv_r2 <- function(form, data, K = params$cv_folds, seed = params$seed) tryCatch({
  vars <- all.vars(form)
  cc <- data[stats::complete.cases(data[, vars, drop = FALSE]), , drop = FALSE]
  if (nrow(cc) < 2 * K) return(NA_real_)
  set.seed(seed); fold <- sample(rep_len(1:K, nrow(cc)))
  pr <- numeric(nrow(cc))
  for (k in 1:K) {
    fit <- stats::lm(form, cc[fold != k, , drop = FALSE])
    pr[fold == k] <- stats::predict(fit, cc[fold == k, , drop = FALSE])
  }
  yv <- cc[[vars[1]]]; 1 - sum((yv - pr)^2) / sum((yv - mean(yv))^2)
}, error = function(e) NA_real_)

mk <- function(terms) reformulate(if (length(terms)) terms else "1", "y")

# Shapley/LMG within the maternal-production block, over the full foundation base.
shapley_lmg <- function(preds, base, data, ycol = "y") {
  vars <- c(ycol, base, preds)
  cc <- data[stats::complete.cases(data[, vars, drop = FALSE]), , drop = FALSE]
  R  <- function(terms) summary(stats::lm(reformulate(if (length(terms)) terms else "1", ycol), cc))$r.squared
  perms <- function(x) if (length(x) <= 1) list(x) else
    do.call(c, lapply(seq_along(x), function(i) lapply(perms(x[-i]), function(p) c(x[i], p))))
  P <- perms(preds); contrib <- stats::setNames(numeric(length(preds)), preds)
  for (p in P) { prev <- base
    for (v in p) { contrib[v] <- contrib[v] + (R(c(prev, v)) - R(prev)); prev <- c(prev, v) } }
  contrib / length(P)
}
# block labels, in ladder order (design/clinical foundation split three ways)
BLOCK_LEVELS <- c("design", "+ clinical (pre-vaccination)", "+ clinical (birth)",
                  "+ maternal production", "+ cord (transfer)", "+ concurrent (ceiling)")

build_frame <- function(assay) {
  yw <- pull_wide(data_raw, params$visit_outcome, assay, "")
  names(yw)[names(yw) == assay] <- "y"
  m <- yw %>%
    dplyr::left_join(mat, "subject_accession") %>%
    dplyr::left_join(crd, "subject_accession") %>%
    dplyr::left_join(inf, "subject_accession") %>%
    dplyr::left_join(arm, "subject_accession")
  if (!is.null(clind)) m <- dplyr::left_join(m, clind, "subject_accession")
  dplyr::filter(m, !is.na(y))
}

# one outcome -> the six cumulative CV-R2 rungs + the maternal LMG breakdown
run_outcome <- function(assay, label) {
  m <- build_frame(assay)
  pres <- function(cols) cols[cols %in% names(m)]
  design <- pres(DESIGN); prevax <- pres(CLIN_PREVAX); birth <- pres(CLIN_BIRTH)
  mp <- pres(paste0(ANTI_IGG, "_mat"))
  cp <- pres(paste0(ANTI_IGG, "_cord"))
  ip <- pres(paste0(ANTI_IGG, "_inf"))

  # cumulative term sets for the six rungs
  cum <- list(design,
              c(design, prevax),
              c(design, prevax, birth),
              c(design, prevax, birth, mp),
              c(design, prevax, birth, mp, cp),
              c(design, prevax, birth, mp, cp, ip))
  ladder <- data.frame(
    outcome = label,
    block   = factor(BLOCK_LEVELS, levels = BLOCK_LEVELS),
    n       = vapply(cum, function(t) sum(stats::complete.cases(m[, all.vars(mk(t)), drop = FALSE])), integer(1)),
    cv_r2   = vapply(cum, function(t) cv_r2(mk(t), m), numeric(1)),
    stringsAsFactors = FALSE)
  ladder[["delta"]] <- c(NA, diff(ladder$cv_r2))

  # maternal-production antigen shares (LMG) over the full foundation base
  base <- c(design, prevax, birth)
  lmg  <- if (length(mp) >= 2) shapley_lmg(mp, base, m) else stats::setNames(numeric(0), character(0))
  lmg_df <- if (length(lmg))
    data.frame(outcome = label,
               antigen = sub("_IgG_mat$", "", names(lmg)),
               lmg_r2  = as.numeric(lmg), row.names = NULL) else NULL
  list(ladder = ladder, lmg = lmg_df)
}

Figure 5 (revised). Where predictability of infant whole-organism function enters. Left: cumulative cross-validated R² for a six-block ladder — design (vaccination type), + pre-vaccination clinical, + birth clinical, + maternal production (maternal antigen at delivery), + cord transfer, + concurrent (two-month co-measured ceiling). The concurrent rung is a descriptive ceiling, not forward prediction. Right: within-outcome Shapley/LMG variance shares of the maternal-production antigens; the labelled antigen is the principal (largest-share) predictor of each outcome. Both panels exclude the concurrent block from any predictive claim.

Revised Figure 5. The design/clinical foundation of the block ladder is split into three ordered blocks — design (vaccination type), pre-vaccination clinical (maternal age, BMI, haemoglobin, parity, gestational age at vaccination), and birth clinical (gestational age at birth, delivery mode, birth weight, infant sex, vaccine-to-birth interval) — so the six rungs are design, + pre-vaccination clinical, + birth clinical, + maternal production, + cord transfer, + concurrent ceiling. The former “maternal (forward)” block is relabelled “maternal production” throughout, and the right-panel axis reads “maternal production LMG R²”. Adjust the three *_vars params to move a covariate between the pre-vaccination and birth clinical blocks.
